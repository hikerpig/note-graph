# NoteGraphModel

A data model for the notes.

```ts
export type Note = {
  id: string
  title: string
  linkTo?: NodeId[]
  referencedBy?: NodeId[]
  /**
   * You can specify a different type (e.g. 'tag') for a note node.
   * Later you can specify different node style to it.
   */
  type?: string
}

class NoteGraphModel {
  constructor(notes: Note[])
}
```
